  • Raster Source mit einer operation
function flood(pixels, data) {
  const pixel = pixels[0];
  if (pixel[3]) {
    const height =
      -10000 + (pixel[0] * 256 * 256 + pixel[1] * 256 + pixel[2]) * 0.1;
    if (height <= data.level) {
      pixel[0] = 134; pixel[1] = 203; pixel[2] = 249; pixel[3] = 255;
    } else {
      pixel[3] = 0;
    }
  }
  return pixel;
}
